        // --- Audio Context for Beeps ---

        /**

         * Plays a simple beep sound.

         * @param {number} frequency - Frequency of the beep (e.g., 880 Hz).

         * @param {number} duration - Duration in milliseconds (e.g., 100 ms).

         * @param {number} volume - Volume (0.0 to 1.0).

         */

        function playBeep(frequency = 880, duration = 100, volume = 0.1) {

            if (!audioContext) return;

            try {

                const oscillator = audioContext.createOscillator();

                const gainNode = audioContext.createGain();


                oscillator.connect(gainNode);

                gainNode.connect(audioContext.destination);


                oscillator.type = 'sine';

                oscillator.frequency.setValueAtTime(frequency, audioContext.currentTime);

                gainNode.gain.setValueAtTime(volume, audioContext.currentTime);


                oscillator.start(audioContext.currentTime);

                oscillator.stop(audioContext.currentTime + duration / 1000);

            } catch (e) {

                console.warn("Error playing beep:", e);

            }

        }

        // --- Timer Logic ---


        /**

         * Formats seconds into a MM:SS string.

         * @param {number} seconds - The number of seconds.

         * @returns {string} - The formatted time string.

         */

        function formatTime(seconds) {

            const m = Math.floor(seconds / 60);

            const s = seconds % 60;

            return `${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;

        }


        /**

         * Stops the currently running timer interval.

         */

        function stopTimer() {

            clearInterval(timerInterval);

            timerInterval = null;

        }


        /**

         * Updates the timer clock every second.

         */

        function timerTick() {

            totalSeconds--;

            timerClock.textContent = formatTime(totalSeconds);


            if (totalSeconds <= 0) {

                stopTimer();

                // Check if the *entire* test is over

                const isTestOver = (

                    currentTest === 'srt' || 

                    currentTest === 'sdt' ||

                    (currentTest === 'ppdt' && testState.phase === 'write') ||

                    (currentTest === 'tat' && testState.currentPicture > 12) ||

                    (currentTest === 'wat' && testState.currentWord > 60) ||

                    (currentTest === 'lecturette' && testState.phase === 'speak')

                );


                if (isTestOver) {

                    playBeep(1200, 200, 0.1); // Higher beep for test end

                    handleTestCompletion();

                } else {

                    playBeep(880, 100, 0.1); // Standard beep for phase change

                    runTestTick(); // Move to the next phase

                }

            }

        }


        /**

         * Starts a timer for a given number of seconds.

         * @param {number} seconds - The duration of the timer.

         */

        function startTimer(seconds) {

            stopTimer();

            totalSeconds = seconds;

            timerClock.textContent = formatTime(totalSeconds);

            timerInterval = setInterval(timerTick, 1000);

        }


        /**

        * Handles the main logic for starting a selected test.

        */

        function startTest() {

            if (!currentTest) return;


            // Hide welcome screens and show timer

            hideAllScreens();

            timerDisplay.classList.remove('hidden');


            // Toggle footer buttons

            startButton.classList.add('hidden');

            resetButton.classList.remove('hidden');


            // Disable tabs

            testTabs.forEach(tab => {

                tab.disabled = true;

                tab.classList.add('tab-disabled');

            });


            // Initialize test-specific state

            switch (currentTest) {

                case 'ppdt':

                    testState = { phase: 'observe' };

                    break;

                case 'tat':

                    testState = { currentPicture: 1, phase: 'observe' };

                    break;

                case 'wat':

                    // Shuffle words for this session

                    shuffledWatWords = [...watWords].sort(() => Math.random() - 0.5);

                    testState = { currentWord: 1 };

                    watWordDisplay.classList.remove('hidden');

                    break;

                case 'srt':

                    testState = {};

                    break;

                case 'sdt':

                    testState = {};

                    break;

                case 'lecturette':

                    testState = { phase: 'prep' };

                    break;

            }


            runTestTick(); // Start the first phase

        }


        /**

         * Runs the next step/phase of the currently active test.

         */

        function runTestTick() {

            switch (currentTest) {

                case 'ppdt':

                    if (testState.phase === 'observe') {

                        // Start 'Observe' phase

                        timerPhase.textContent = 'Observe Picture';

                        timerProgress.textContent = 'PPDT - Screening Test';

                        tatImagePlaceholder.classList.remove('hidden');

                        testState.phase = 'write';

                        startTimer(30); // 30 seconds

                    } else {

                        // Start 'Write' phase

                        timerPhase.textContent = 'Write Story';

                        timerProgress.textContent = 'PPDT - Screening Test';

                        tatImagePlaceholder.classList.add('hidden');

                        startTimer(240); // 4 minutes

                    }

                    break;


                case 'tat':

                    if (testState.currentPicture > 12) {

                        handleTestCompletion();

                        return;

                    }

                    if (testState.phase === 'observe') {

                        // Start 'Observe' phase

                        timerPhase.textContent = testState.currentPicture === 12 ? 'Prepare (Blank Slide)' : 'Observe Picture';

                        timerProgress.textContent = `Picture ${testState.currentPicture} of 12`;

                        tatImagePlaceholder.classList.toggle('hidden', testState.currentPicture === 12);

                        testState.phase = 'write';

                        startTimer(30); // 30 seconds

                    } else {

                        // Start 'Write' phase

                        timerPhase.textContent = 'Write Story';

                        timerProgress.textContent = `Picture ${testState.currentPicture} of 12`;

                        tatImagePlaceholder.classList.add('hidden');

                        testState.phase = 'observe';

                        testState.currentPicture++;

                        startTimer(240); // 4 minutes

                    }

                    break;


                case 'wat':

                    if (testState.currentWord > 60) {

                        handleTestCompletion();

                        return;

                    }

                    // Display current word

                    const word = shuffledWatWords[testState.currentWord - 1];

                    watWordText.textContent = word;

                    

                    // Update UI

                    timerPhase.textContent = 'Write Sentence';

                    timerProgress.textContent = `Word ${testState.currentWord} of 60`;

                    

                    // Move to next word

                    testState.currentWord++;

                    startTimer(15); // 15 seconds

                    break;


                case 'srt':

                    timerPhase.textContent = 'Situation Reaction Test';

                    timerProgress.textContent = 'Time Remaining';

                    startTimer(1800); // 30 minutes

                    break;


                case 'sdt':

                    timerPhase.textContent = 'Self-Description Test';

                    timerProgress.textContent = 'Time Remaining';

                    startTimer(900); // 15 minutes

                    break;

                

                case 'lecturette':

                    if (testState.phase === 'prep') {

                        // Start 'Prep' phase

                        timerPhase.textContent = 'Preparation';

                        timerProgress.textContent = 'Prepare your topic';

                        testState.phase = 'speak';

                        startTimer(180); // 3 minutes

                    } else {

                        // Start 'Speak' phase

                        timerPhase.textContent = 'Speak';

                        timerProgress.textContent = 'Deliver your lecturette';

                        startTimer(180); // 3 minutes

                    }

                    break;

            }

        }


        /**

         * Handles the completion of a test.

         */

        function handleTestCompletion() {

            stopTimer(); // Ensure timer is stopped

            timerPhase.textContent = 'Test Complete!';

            timerProgress.textContent = 'Well done.';

            timerClock.textContent = '00:00';

            

            // Hide test-specific UI

            tatImagePlaceholder.classList.add('hidden');

            watWordDisplay.classList.add('hidden');

        }


        // --- Lecturette Generation Logic ---


        /**

         * Formats the structured JSON response from Gemini into HTML.

         * @param {object} structuredResponse - The parsed JSON object.

         * @returns {string} - The HTML string to display.

         */

        function formatLecturettePoints(structuredResponse) {

            let html = '';

            

            const createSection = (title, points) => {

                if (!points || points.length === 0) return '';

                let sectionHtml = `<h4 class="text-lg font-semibold text-purple-600 dark:text-purple-400 mt-3 mb-2">${title}</h4><ul>`;

                points.forEach(point => {

                    // Check for Concept: Elaboration format

                    if (point.includes(':')) {

                        const parts = point.split(':', 2);

                        sectionHtml += `<li class="ml-4 list-disc marker:text-purple-400"><strong>${parts[0]}:</strong> ${parts[1]}</li>`;

                    } else {

                        sectionHtml += `<li class="ml-4 list-disc marker:text-purple-400">${point}</li>`;

                    }

                });

                sectionHtml += '</ul>';

                return sectionHtml;

            };


            html += createSection('Introduction', structuredResponse.introduction);

            html += createSection('Body', structuredResponse.body);

            html += createSection('Conclusion', structuredResponse.conclusion);

            

            return html;

        }


        /**

         * Fetches lecturette points from the Gemini API.

         */

        async function generateLecturettePoints() {

            const topic = lecturetteTopicInput.value;

            if (!topic) {

                lecturetteError.textContent = 'Please enter a topic first.';

                lecturetteError.classList.remove('hidden');

                return;

            }


            // Set loading state

            lecturetteGenerateButton.disabled = true;

            generateText.textContent = 'Generating...';

            generateSpinner.classList.remove('hidden');

            lecturettePointsContainer.classList.add('hidden');

            lecturetteError.classList.add('hidden');


            const systemPrompt = `

                You are an assistant for an SSB (Services Selection Board) aspirant.

                Your task is to generate preparatory points for a 3-minute lecturette on a given topic.

                The points MUST be structured into "Introduction", "Body", and "Conclusion".

                The "Body" section should contain the main content.

                Each point MUST be a single, short, and simple foundational idea, formatted as "Concept: Elaboration".

                The user will build their own speech based on these simple ideas.

                

                Respond ONLY with a valid JSON object in the following format:

                {

                  "introduction": ["Point 1", "Point 2"],

                  "body": ["Concept 1: Elaboration 1", "Concept 2: Elaboration 2", "Concept 3: Elaboration 3", ...],

                  "conclusion": ["Point 1", "Point 2"]

                }

            `;


            const payload = {

                contents: [{ parts: [{ text: topic }] }],

                systemInstruction: {

                    parts: [{ text: systemPrompt }]

                },

                generationConfig: {

                    responseMimeType: "application/json",

                    responseSchema: {

                        type: "OBJECT",

                        properties: {

                            "introduction": {

                                type: "ARRAY",

                                items: { type: "STRING" }

                            },

                            "body": {

                                type: "ARRAY",

                                items: { type: "STRING" }

                            },

                            "conclusion": {

                                type: "ARRAY",

                                items: { type: "STRING" }

                            }

                        },

                        required: ["introduction", "body", "conclusion"]

                    }

                }

            };

            

            let retries = 3;

            let success = false;

            while (retries > 0 && !success) {

                try {

                    const response = await fetch(API_URL + API_KEY, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    if (!response.ok) {

                        throw new Error(`API Error: ${response.statusText}`);

                    }


                    const result = await response.json();

                    

                    if (result.candidates && result.candidates[0].content && result.candidates[0].content.parts[0].text) {

                        const jsonText = result.candidates[0].content.parts[0].text;

                        const structuredResponse = JSON.parse(jsonText);

                        

                        // Format and display points

                        lecturettePointsContainer.innerHTML = formatLecturettePoints(structuredResponse);

                        lecturettePointsContainer.classList.remove('hidden');

                        success = true;


                    } else {

                        throw new Error("Invalid response structure from API.");

                    }


                } catch (error) {

                    console.error("Error generating lecturette points:", error);

                    retries--;

                    if (retries === 0) {

                        lecturetteError.textContent = 'Oops! Could not generate points. Please check the console or try again.';

                        lecturetteError.classList.remove('hidden');

                    } else {

                        // Exponential backoff wait

                        await new Promise(res => setTimeout(res, (3 - retries) * 1000));

                    }

                }

            }


            // Reset button state

            lecturetteGenerateButton.disabled = false;

            generateText.textContent = 'Generate';

            generateSpinner.classList.add('hidden');

        }
